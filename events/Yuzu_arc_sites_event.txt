fleet_event = {
	id = Yuzu_arc_sites_event.6
	title = Yuzu_arc_sites_event.6.name
	desc = {
		text = Yuzu_arc_sites_event.6.desc.first_finder
		trigger = {
			NOT = { has_global_flag = yuzu_old_abyss_system_discovered }
		}
	}
	desc = {
		text = Yuzu_arc_sites_event.6.desc.not_first_finder
		trigger = {
			has_global_flag = yuzu_old_abyss_system_discovered
		}
	}
	picture = GFX_evt_senrenbanka_field
	#show_sound = event_finding_loot
	show_sound = event_default
	location = from

	archaeology = yes

	is_triggered_only = yes

	option = {
		name = REMARKABLE
		owner = {
			if = {
				limit = {
					NOR = {
						has_technology = tech_yuzu_space_time_theory
						has_tech_option = tech_yuzu_space_time_theory
					}
				}
				add_research_option = tech_yuzu_space_time_theory
				add_tech_progress = {
					tech = tech_yuzu_space_time_theory
					progress = 0.25
				}
			}
			if = {
				limit = {
					NOT = { has_global_flag = yuzu_old_abyss_system_discovered }
				}
				set_global_flag = yuzu_old_abyss_system_discovered
				from.solar_system = {
					spawn_system = {
						min_distance = 20
						max_distance = 30
						max_jumps = 0
						initializer = "yuzu_old_abyss_system"
					}
					if = {
						limit = { last_created_system = { NOT = { has_hyperlane_to = prev } } }
						add_hyperlane = {
							from = this
							to = last_created_system
						}
					}
					else = {
						add_extra_hyperlane_to_spawned_system_effect = yes
					}
				}
				hidden_effect = {
					country_event = { id = Yuzu_window_event.1 days = 1 }
				}
			}
			else = {
				hidden_effect = {
					country_event = { id = Yuzu_window_event.2 days = 1 }
				}
			}
		}
	}
}
