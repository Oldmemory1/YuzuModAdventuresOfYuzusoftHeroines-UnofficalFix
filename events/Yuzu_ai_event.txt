namespace = Yuzu_ai_event

country_event = {
	id = Yuzu_ai_event.1
	hide_window = yes

	is_triggered_only = yes

	trigger = {
		is_ai = yes
		is_YUZU = yes
	}

	immediate = {
		every_owned_planet = {
			limit = {
				has_building = building_yuzu_clone_vats
				OR = {
					has_building = building_robot_assembly_plant
					has_building = building_machine_assembly_plant
					has_building = building_machine_assembly_complex
				}
			}
			remove_building = building_robot_assembly_plant
			remove_building = building_machine_assembly_plant
			remove_building = building_machine_assembly_complex
		}
	}
}

# 舰队增援
country_event = {
	id = Yuzu_ai_event.6
	hide_window = yes

	is_triggered_only = yes

	trigger = {
		is_ai = yes
		is_YUZU = yes
		has_country_resource = {
			type = fragment_of_heart
			amount >= 1000
		}
		OR = {
			has_monthly_income = { resource = energy value >= 50 }
			has_country_resource = {
				type = energy
				amount >= 5000
			}
		}
		OR = {
			has_monthly_income = { resource = alloys value >= 10 }
			has_country_resource = {
				type = alloys
				amount >= 1000
			}
		}
		OR = {
			AND = {
				used_naval_capacity_percent < 0.20
				max_naval_capacity > 150
			}
			AND = {
				used_naval_capacity_percent < 0.50
				max_naval_capacity > 300
			}
			AND = {
				used_naval_capacity_percent < 0.80
				max_naval_capacity > 600
			}
		}
	}

	immediate = {
		if = {
			limit = {
				has_ethic = ethic_fanatic_spiritualist
			}
			add_resource = { fragment_of_heart = -800 }
		}
		else_if = {
			limit = {
				has_ethic = ethic_spiritualist
			}
			add_resource = { fragment_of_heart = -900 }
		}
		else = {
			add_resource = { fragment_of_heart = -1000 }
		}
		country_event = { id = Yuzu_country_event.104 }
	}
}
