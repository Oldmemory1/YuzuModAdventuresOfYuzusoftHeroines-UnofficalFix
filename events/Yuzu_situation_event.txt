planet_event = {
	id = Yuzu_situation_event.1
	title = Yuzu_situation_event.1.name
	desc = Yuzu_situation_event.1.desc
	picture = GFX_evt_yuzu_slime
	show_sound = event_alien_nature #event_wind_ruins
	is_triggered_only = yes

	location = root

	pre_triggers = {
		has_owner = yes
		original_owner = yes
		has_ground_combat = no
		is_occupied_flag = no
	}

	trigger = {
		exists = owner
		owner = {
			is_YUZU = yes
		}
	}

	immediate = {
		owner = {
			start_situation = {
				type = situation_yuzu_slime
				target = root
				effect = {
					save_event_target_as = yuzu_slime_situation
				}
			}
		}
	}

	option = {
		name = Yuzu_situation_event.1.a
		icon = {
			icon = GFX_situation_approach_shrug
			icon_background = GFX_situation_approach_bg_yellow
			text = set_situation_approach_icon_text
		}
		event_target:yuzu_slime_situation = {
			set_situation_approach = approach_yuzu_do_nothing
		}
		ai_chance = {
			factor = 2
			modifier = {
				factor = 3
				owner = {
					OR = {
						is_pacifist = yes
						is_xenophile = yes
					}
				}
			}
		}
	}

	option = {
		name = Yuzu_situation_event.1.b
		icon = {
			icon = GFX_situation_approach_sword
			icon_background = GFX_situation_approach_bg_red
			text = set_situation_approach_icon_text
		}
		event_target:yuzu_slime_situation = {
			set_situation_approach = approach_yuzu_eliminate
		}
		ai_chance = {
			factor = 2
			modifier = {
				factor = 3
				owner = {
					OR = {
						is_militarist = yes
						is_xenophobe = yes
					}
				}
			}
		}
	}
}
