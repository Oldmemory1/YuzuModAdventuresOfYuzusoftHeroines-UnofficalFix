	id = Yuzu_country_event.1

		has_origin = origin_yuzu_world
	}

	immediate = {
		every_system_within_border = {
			every_fleet_in_system = {
				limit = {
					NOR = {
						is_ship_class = shipclass_mining_station
						is_ship_class = shipclass_research_station
						is_ship_class = shipclass_observation_station
						is_ship_class = shipclass_military_station
						AND = {
							is_ship_class = shipclass_starbase
							NOT = { is_ship_size = juggernaut }
						}
					}
					controller = {
						is_hostile = root
					}
				}
				if = {
					limit = {
						has_fleet_flag = Presiding_Judge_Fleet
					}
					set_global_flag = murasamemuru_eliminated_abyss
					root = {
						country_event = { id = Yuzu_abyss_crisis_event.107 days = 1 }
					}
				}
				destroy_fleet = this
			}
		}
	}

	option = {
		name = Yuzu_country_event.1.a
	}
}

country_event = {
	id = Yuzu_country_event.20
	hide_window = yes
	is_triggered_only = yes

	immediate = {
		if = {
			limit = {
				exists = overlord
				overlord = { has_valid_civic = civic_yuzu_win_the_hearts }
			}
			remove_modifier = yuzu_win_the_hearts
			if = {
				limit = {
					overlord = {
						has_councilor = { COUNCILOR = councilor_yuzu_win_the_hearts }
					}
				}
				add_modifier = {
					modifier = yuzu_win_the_hearts
					mult = value:yuzu_overlord_councilor_level|TYPE|councilor_yuzu_win_the_hearts|
				}
			}
			else = {
				add_modifier = {
					modifier = yuzu_win_the_hearts
				}
			}
			if = {
				limit = {
					NOT = { has_opinion_modifier = { who = overlord modifier = opinion_yuzu_win_the_hearts } }
				}
				add_opinion_modifier = {
					who = overlord
					modifier = opinion_yuzu_win_the_hearts
				}
			}
			else_if = {
				limit = {
					NOT = { has_opinion_modifier = { who = overlord modifier = opinion_yuzu_fanatic } }
				}
				random_list = {
					1 = {
						add_opinion_modifier = {
							who = overlord
							modifier = opinion_yuzu_fanatic
						}
					}
					99 = { }
				}
			}
		}
		else_if = {
			limit = { has_modifier = yuzu_win_the_hearts }
			remove_modifier = yuzu_win_the_hearts
		}
	}
}

country_event = {
	id = Yuzu_country_event.105
	title = Yuzu_country_event.105.name
	desc = "Yuzu_country_event.105.desc"

	picture = GFX_evt_yuzu_meguru
	show_sound = event_mind_over_matter

	is_triggered_only = yes

	immediate = {
		every_country = {
			limit = {
				OR = {
					is_country_type = default
				    is_country_type = fallen_empire
				    is_country_type = awakened_fallen_empire
				}
			}
			country_event = {
				id = Yuzu_country_event.106
			}
		}
		every_war = {
			end_war_effect = yes
		}
		if = {
			limit = {
				has_global_flag = war_in_heaven_ongoing
			}
			remove_global_flag = war_in_heaven_ongoing
			set_global_flag = war_in_heaven_ended
		}
	}

	option = {
		name = OK
	}
}
