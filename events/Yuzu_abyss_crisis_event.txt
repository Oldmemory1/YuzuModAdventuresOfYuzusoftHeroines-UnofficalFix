					from = this
					to = last_created_system
				}
			}
			else = {
				add_extra_hyperlane_to_spawned_system_effect = yes
			}
		}
		country_event = { id = Yuzu_abyss_crisis_event.11 }

			limit = {
				OR = {
					is_country_type = default
