# 堕天神兵建造完成
ship_event = {
	id = Yuzu_window_event.5
	title = Yuzu_window_event.5.name
	desc = Yuzu_window_event.5.desc
	picture = GFX_evt_binary_stars
	show_sound = event_default

	is_triggered_only = yes

	trigger = {
		is_ship_size = yuzu_star_eater
		owner = {
			NOT = { has_country_flag = yuzu_star_eater_constructed }
		}
	}

	immediate = {
		owner = {
			set_country_flag = yuzu_star_eater_constructed
		}
	}

	option = {
		name = Yuzu_window_event.5.a
		trigger = {
			owner = {
				NOR = {
					has_ethic = ethic_pacifist
					has_ethic = ethic_fanatic_pacifist
					has_ethic = ethic_xenophile
					has_ethic = ethic_fanatic_xenophile
				}
			}
		}
	}
	option = {
		name = Yuzu_window_event.5.b
		trigger = {
			owner = {
				OR = {
					has_ethic = ethic_pacifist
					has_ethic = ethic_fanatic_pacifist
					has_ethic = ethic_xenophile
					has_ethic = ethic_fanatic_xenophile
				}
			}
		}
	}
}
