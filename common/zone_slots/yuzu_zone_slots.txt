# Yuzu City World Zone Slots
# Same structure as vanilla slot_city_01/02 but with restricted zone availability
# for Yuzu city worlds (abyss / sunset).
#
# Excluded from vanilla urban set:
#   zone_spawning           - hive only
#   zone_machine_replication - machine world only (pc_machine)
#
# Compared to vanilla district_arcology_housing, the Yuzu housing district uses
# these slots instead of slot_city_01/02 to curate the available specializations.

slot_yuzu_city_01 = {
	included_zone_sets = {
		urban
	}

	exclude = {
		zone_spawning
		zone_machine_replication
	}

	unlock = {
		OR = {
			uses_district_set = yuzu_abyss_city_world
			uses_district_set = yuzu_sunset_city_world
		}
	}
}

slot_yuzu_city_02 = {
	included_zone_sets = {
		urban
	}

	exclude = {
		zone_spawning
		zone_machine_replication
	}

	unlock = {
		OR = {
			uses_district_set = yuzu_abyss_city_world
			uses_district_set = yuzu_sunset_city_world
		}
		custom_tooltip = {
			fail_text = zone_city_02_prereq
			if = {
				limit = {
					exists = owner
					owner = { is_primitive = yes }
				}
				always = yes
			}
			else = {
				has_upgraded_capital = yes
			}
		}
	}
}
