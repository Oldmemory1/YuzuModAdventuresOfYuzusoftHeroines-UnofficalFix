	set_yuzu_city_district = {
		while = {
			count = num_district_housing_yuzu
			add_district = district_yuzu_city_housing
		}
		while = {
			count = num_district_arms_industry_yuzu
			add_district = district_yuzu_city_industry
		}
		while = {
			count = num_district_civilian_industry_yuzu
			add_district = district_yuzu_city_industry
		}
		while = {
			count = num_district_leisure_yuzu
			add_district = district_yuzu_city_housing
		}
		while = {
			count = num_district_administrative_yuzu
			add_district = district_yuzu_city_administrative
		}
		while = {
			count = num_district_religious_yuzu
			add_district = district_yuzu_city_administrative
		}
		clear_variable = num_district_housing_yuzu
		clear_variable = num_district_arms_industry_yuzu
		clear_variable = num_district_civilian_industry_yuzu
		clear_variable = num_district_leisure_yuzu

	yuzu_store_pops = {
		owner = {
			set_variable = { which = num_lacked_yuzu_pops value = 10 }
			save_event_target_as = yuzu_planet_owner
		}
		while = {
			count = 10
			random_owned_pop_group = {
				limit = { is_YUZU = yes }
				owner = {
					subtract_variable = { which = num_lacked_yuzu_pops value = 1 }
				}
				kill_single_pop = yes
			}
		}
		if = {  # 如果建造完成时星球上人口不足，从整个国家中减去缺少的，防止刷人口
			limit = {
				event_target:yuzu_planet_owner = {
					check_variable = { which = num_lacked_yuzu_pops value >= 1 }
				}
			}
			event_target:yuzu_planet_owner = {
				while = {
					count = num_lacked_yuzu_pops
					random_owned_pop_group = {
						limit = { is_YUZU = yes }
						kill_single_pop = yes
					}
				}
			}
		}
		event_target:yuzu_planet_owner = {
			clear_variable = num_lacked_yuzu_pops
		}
	}

	yuzu_release_pops = {
		if = {
			limit = {
				owner = { is_YUZU = yes }
			}
			create_pop_group = {
				species = owner_main_species
				size = 1000
				effect = {
					add_modifier = {
						modifier = pop_yuzu_released_from_cryo_chamber
						days = 3600
					}
				}
			}
		}
		else_if = {
			limit = {
				any_galaxy_species = { is_species_class = YUZUGIRL }
			}
			random_galaxy_species = {
				limit = {
					is_species_class = YUZUGIRL
				}
				save_event_target_as = yuzu_released_species
			}
			create_pop_group = {
				species = event_target:yuzu_released_species
				size = 1000
				effect = {
					add_modifier = {
						modifier = pop_yuzu_released_from_cryo_chamber
						days = 3600
					}
				}
			}
		}
		else = {
			create_species = {
				name = "NAME_Yuzu"
				class = YUZUGIRL
				portrait = random
				traits = {
					ideal_planet_class = pc_continental
					trait = trait_yuzu
					trait = trait_yuzu_ciallo
					trait = trait_intelligent
					trait = trait_traditional
					trait = trait_talented
					trait = trait_communal
					trait = trait_yuzu_big_eater
					trait = trait_weak
					trait = trait_sedentary
				}
				effect = { save_event_target_as = yuzu_released_species }
			}
			create_pop_group = {
				species = event_target:yuzu_released_species
				size = 1000
				effect = {
					add_modifier = {
						modifier = pop_yuzu_released_from_cryo_chamber
						days = 3600
					}
				}
			}
		}
	}
