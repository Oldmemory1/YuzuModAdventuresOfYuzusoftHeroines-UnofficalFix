add_yuzu_leader_trait_yuzu = {
	if = {
		limit = {
			species = {
				has_trait = trait_yuzu
			}
			NOT = { has_trait = leader_trait_immortal_yuzu }
		}
		add_trait = leader_trait_yuzu
	}
}

remove_yuzu_leader_trait_yuzu = {
	if = {
		limit = {
			has_trait = leader_trait_yuzu
		}
		remove_trait = leader_trait_yuzu
	}
}

set_yuzu_leader_skill_single = {
	#split
	if = {
		limit = {
			owner = { check_variable = { which = yuzu_$LEADER_CLASS$_skill value = 10 } }
		}
		while = {
			count = 8
			if = {
				limit = { has_total_skill < 9 }
				add_skill_without_trait_selection = 1
			}
		}
	}
	else_if = {
		limit = {
			owner = { check_variable = { which = yuzu_$LEADER_CLASS$_skill value = 9 } }
		}
		while = {
			count = 7
			if = {
				limit = { has_total_skill < 8 }
				add_skill_without_trait_selection = 1
			}
		}
	}
	else_if = {
		limit = {
			owner = { check_variable = { which = yuzu_$LEADER_CLASS$_skill value = 8 } }
		}
		while = {
			count = 6
			if = {
				limit = { has_total_skill < 7 }
				add_skill_without_trait_selection = 1
			}
		}
	}
	else_if = {
		limit = {
			owner = { check_variable = { which = yuzu_$LEADER_CLASS$_skill value = 7 } }
		}
		while = {
			count = 5
			if = {
				limit = { has_total_skill < 6 }
				add_skill_without_trait_selection = 1
			}
		}
	}
	else_if = {
		limit = {
			owner = { check_variable = { which = yuzu_$LEADER_CLASS$_skill value = 6 } }
		}
		while = {
			count = 4
			if = {
				limit = { has_total_skill < 5 }
				add_skill_without_trait_selection = 1
			}
		}
	}
	else_if = {
		limit = {
			owner = { check_variable = { which = yuzu_$LEADER_CLASS$_skill value = 5 } }
		}
		while = {
			count = 3
			if = {
				limit = { has_total_skill < 4 }
				add_skill_without_trait_selection = 1
			}
		}
	}
	else_if = {
		limit = {
			owner = { check_variable = { which = yuzu_$LEADER_CLASS$_skill value = 4 } }
		}
		while = {
			count = 2
			if = {
				limit = { has_total_skill < 3 }
				add_skill_without_trait_selection = 1
			}
		}
	}
	else_if = {
		limit = {
			owner = { check_variable = { which = yuzu_$LEADER_CLASS$_skill value = 3 } }
		}
		if = {
			limit = { has_total_skill < 2 }
			add_skill_without_trait_selection = 1
		}
	}
	else_if = {
		limit = {
			owner = { check_variable = { which = yuzu_$LEADER_CLASS$_skill value = 2 } }
		}
		if = {
			limit = { has_total_skill < 2 }
			add_skill_without_trait_selection = 1
		}
	}
}

	every_owned_leader = {
		limit = {
			leader_class = $LEADER_CLASS$
		}
		set_yuzu_leader_skill_single = { LEADER_CLASS = $LEADER_CLASS$ }
	}
	every_pool_leader = {
		limit = {
			leader_class = $LEADER_CLASS$
