create_presiding_judge_fleet_and_constructors = {
	if = {
		limit = {
			NOT = { has_global_flag = Presiding_Judge_Fleet_Created }
			owner = { is_country_type = new_abyss_empire }
		}
		set_global_flag = Presiding_Judge_Fleet_Created
		owner = {
			create_new_abyss_fleet_3 = yes
			while = {
				count = 3
				create_new_abyss_constructor = yes
			}
		}
		every_country = {
			limit = {
				OR = {
					is_country_type = default
					is_country_type = fallen_empire
					is_country_type = awakened_fallen_empire
				}
			}
			country_event = { id = Yuzu_abyss_crisis_event.109 }
		}
	}
